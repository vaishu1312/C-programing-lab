void count(FILE *fp)
{
  struct ticket t;
  rewind(fp);
  int record = 0;
  while(!feof(fp))
  {
    record = fread(&t, sizeof(t), 1, fp);
    if(record && strcmp(t.name,""))
    {
      if (t.Class == A3) A++;
      else if (t.Class == FC) F++;
      else if (t.Class == SL) S++;
    } 
  } // end-while
}

void book(FILE *fp)
{
  struct ticket t;
  printf("Enter name: ");
  scanf("%s", t.name);
  printf("Enter mobile number: ");
  scanf("%d", &t.number);
  printf("Enter age: ");
  scanf("%d", &t.age);
  printf("Enter Class (AC - 0, FC - 1, SL - 2): ");
  scanf("%d",&t.Class); 
  if (t.Class == A3) A++;
  else if (t.Class == FC) F++;
  else if (t.Class == SL) S++;
  fwrite(&t, sizeof(struct ticket), 1, fp);
}

    {
      fseek(fp, record*sizeof(struct ticket), SEEK_SET);
